'use strict'

/** @module Core */

/**
 * @memberof module:Core
 */
function Mapping (name, body) {
  this.WKName = name
  this.body = body
}

/**
 * Provides a JSON version of the Mapping
 * @returns {*}
 */
Mapping.prototype.toJson = function () {
  let p = JSON.parse(JSON.stringify(this))
  delete p.WKName
  return p.body
}

module.exports = {
  Mapping: Mapping
}
